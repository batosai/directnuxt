# Pour Directus
cd workspaces
npx create-directus-project directus
cd directus
npm install

# Pour Nuxt
cd ..
npx nuxi init nuxt-app
cd nuxt-app
npm install

# Pour Storybook
cd ..
npx storybook init storybook
cd storybook
npm install
